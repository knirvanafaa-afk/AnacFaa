const supabase = createClient(supabaseUrl, supabaseKey);

export async function validarAcessoTotal() {
    try {
        // 1. IDENTIDADE (Verifica se existe sessão)
        const { data: { session } } = await supabase.auth.getSession();
        if (!session) {
            window.location.href = 'index.html';
            return false;
        }

        // 2. VERIFICAÇÃO (Email confirmado)
        if (!session.user.email_confirmed_at) {
            alert("Acesso negado: Por favor, confirme seu e-mail para acessar.");
            window.location.href = 'index.html';
            return false;
        }

        // 3. SEGURANÇA ADICIONAL (MFA AAL2)
        const { data: mfaData } = await supabase.auth.mfa.getAuthenticatorAssuranceLevel();
        if (mfaData.currentLevel !== 'aal2') {
            window.location.href = 'verificar-mfa.html';
            return false;
        }

        // 4. AUTORIZAÇÃO (Status no banco + Expiração)
        const { data: profile, error } = await supabase
            .from('profiles')
            .select('status_acesso, expira_em')
            .eq('id', session.user.id)
            .single();

        if (error || !profile) {
            window.location.href = 'index.html';
            return false;
        }

        const ehBetaOuPago = (profile.status_acesso === 'beta' || profile.status_acesso === 'pago');
        const estaExpirado = profile.expira_em && new Date(profile.expira_em) < new Date();

        if (!ehBetaOuPago || estaExpirado) {
            alert("Acesso negado: Plano expirado ou sem permissão.");
            window.location.href = 'index.html';
            return false;
        }

        return true;

    } catch (err) {
        console.error("Erro na validação:", err);
        window.location.href = 'index.html';
        return false;
    }
}
